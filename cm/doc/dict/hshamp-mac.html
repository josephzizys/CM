[Macro]
#& name

Returns the object named name. 

cm> (new seq :name 'stuff)
#<seq: "stuff">
cm> #&stuff
#<seq: "stuff">
cm> (io "test.clm")
#<clm-stream: "test.clm">
cm> #&"test.clm"
#<clm-stream: "test.clm">
cm>
